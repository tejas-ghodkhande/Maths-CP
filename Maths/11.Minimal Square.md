# Solution

```C++
#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       int a,b;
       cin>>a>>b;
       if(a>b)
          b=2*b;
       else
          a=2*a;
        int area=max(a,b)*max(a,b);
        cout<<area<<endl;
    }  
    return 0;
}
```
